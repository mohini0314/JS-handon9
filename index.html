abcadeecfb
Output:
a=2
b=2
c=2
d=1
e=2
f=1 

  function CountAlph(STR, N )
  {
    var d = new Map();

    STR.split('').forEach( N => {
      if (d.has(N))
      {
        d.set(N, d.get(N)+1);
      }
      else 
      {
        d.set(N, 1);
      }
    });
    STR.split('').forEach(N =>{
      if(d.has(N) && d.get(N)!=0)
      {
        console.log(N + "=" + d.get(N) + "");
        d.set(N, 0);
      }
    });
  }
